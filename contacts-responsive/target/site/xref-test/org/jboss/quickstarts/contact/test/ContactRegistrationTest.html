95      @Test
96      @InSequence(1)
97      public void testRegister() throws Exception {
98          Contact contact = createContactInstance("Jack", "Doe", "jack@mailinator.com", "2125551234", date);
99          Response response = contactRESTService.createContact(contact);
100 
101         assertEquals("Unexpected response status", 200, response.getStatus());
102         log.info(" New contact was persisted and returned status " + response.getStatus());
103     }

105     @SuppressWarnings("unchecked")
106     @Test
107     @InSequence(2)
108     public void testInvalidRegister() throws Exception {
109         Contact contact = createContactInstance("", "", "", "", date);
110         Response response = contactRESTService.createContact(contact);
111 
112         assertEquals("Unexpected response status", 400, response.getStatus());
113         assertNotNull("response.getEntity() should not be null", response.getEntity());
114         assertEquals("Unexpected response.getEntity(). It contains " + response.getEntity(), 3,
115             ((Map<String, String>) response.getEntity()).size());
116         log.info("Invalid contact register attempt failed with return code " + response.getStatus());
117     }
118 
119     @SuppressWarnings("unchecked")
120     @Test
121     @InSequence(3)
122     public void testDuplicateEmail() throws Exception {
123         // Register an initial user
124         Contact contact = createContactInstance("Jane", "Doe", "jane@mailinator.com", "2125551234", date);
125         contactRESTService.createContact(contact);
126 
127         // Register a different user with the same email
128         Contact anotherContact = createContactInstance("John", "Doe", "jane@mailinator.com", "2133551234", date);
129         Response response = contactRESTService.createContact(anotherContact);
130 
131         assertEquals("Unexpected response status", 409, response.getStatus());
132         assertNotNull("response.getEntity() should not be null", response.getEntity());
133         assertEquals("Unexpected response.getEntity(). It contains" + response.getEntity(), 1,
134             ((Map<String, String>) response.getEntity()).size());
135         log.info("Duplicate contact register attempt failed with return code " + response.getStatus());
136     }

183     private Contact createContactInstance(String firstName, String lastName, String email, String phone, Date birthDate) {
184         Contact contact = new Contact();
185         contact.setFirstName(firstName);
186         contact.setLastName(lastName);
187         contact.setEmail(email);
188         contact.setPhoneNumber(phone);
189         contact.setBirthDate(birthDate);
190         return contact;
191     }
